        ${assimp_INCLUDE_DIR}

    )

        ${target}
        ${DEFAULT_LIBRARIES}
        libc++_shared.so
        ${OpenCV_LIBS}
        ${assimp_LIBS}
        ${META_PROJECT_NAME}::lib-SLProject
